SECTION 1: THE TYPE SYSTEM - PRIMITIVES AND STRUCTURAL TYPES
================================================================================

JavaScript operates with a dynamic, weakly-typed system. Unlike statically-typed languages where variable types are declared and enforced at compile-time, JavaScript determines types at runtime and permits implicit type coercion. This flexibility demands rigorous understanding of type mechanics to avoid subtle bugs.

1.1 THE SEVEN PRIMITIVE TYPES
--------------------------------------------------------------------------------

Primitives are immutable, atomic values stored directly in memory. They are NOT objects and have no methods of their own (though JavaScript temporarily wraps them in object wrappers when methods are accessed).

TYPE 1: UNDEFINED
The undefined type has exactly one value: undefined. It represents the absence of an assigned value.

Mechanical Rules:
- Variables declared but not initialized hold undefined
- Function parameters not provided default to undefined
- Object properties that do not exist return undefined
- Functions without explicit return statements return undefined

Syntax:
let declaredButUnassigned;
console.log(declaredButUnassigned); // undefined

Example - Undefined in Function Context:
function demonstrateUndefined(parameterA, parameterB) {
    // parameterB was not provided when function was called
    console.log(parameterA); // "value_A"
    console.log(parameterB); // undefined
    
    let localVariable;
    console.log(localVariable); // undefined
    
    // No return statement means function returns undefined
}

demonstrateUndefined("value_A");
let result = demonstrateUndefined("value_A");
console.log(result); // undefined


TYPE 2: NULL
The null type has exactly one value: null. It represents the intentional absence of any object value.

Mechanical Rules:
- null must be explicitly assigned
- null is NOT the same as undefined (different types)
- typeof null returns "object" (historical JavaScript bug, never fixed for backward compatibility)

Syntax:
let intentionallyEmpty = null;

Example - Null vs Undefined:
let userProfile = null; // Explicitly set to "no profile"
let userSettings; // Implicitly undefined (not yet initialized)


console.log(typeof userProfile); // "object" (JavaScript quirk)
console.log(typeof userSettings); // "undefined"
console.log(userProfile === null); // true
console.log(userSettings === undefined); // true

// Checking for null or undefined
if (userProfile === null) {
    console.log("Profile explicitly set to null");
}

if (userSettings === undefined) {
    console.log("Settings never initialized");
}

// Common pattern: checking for "nullish" values
if (userProfile == null) {
    // This catches BOTH null and undefined (loose equality)
    console.log("Profile is null or undefined");
}


TYPE 3: BOOLEAN
The boolean type has exactly two values: true and false. Booleans represent logical truth values.

Mechanical Rules:
- Boolean literals are lowercase: true, false
- Any value can be coerced to boolean via Boolean() or double negation !!
- Falsy values: false, 0, -0, 0n, "", null, undefined, NaN
- All other values are truthy (including empty objects {} and arrays [])

Syntax:
let isActive = true;
let isDisabled = false;

Example - Boolean Coercion and Truthiness:
// Explicit conversion to boolean
console.log(Boolean(1)); // true
console.log(Boolean(0)); // false
console.log(Boolean("text")); // true
console.log(Boolean("")); // false
console.log(Boolean({})); // true (empty object is truthy!)
console.log(Boolean([])); // true (empty array is truthy!)

// Double negation coercion (idiomatic JavaScript)
console.log(!!"hello"); // true
console.log(!!0); // false
console.log(!!null); // false

// Truthiness in conditional context
let userInput = "";
if (userInput) {
    console.log("User provided input");
} else {
    console.log("Input is empty"); // This executes
}


TYPE 4: NUMBER
The number type represents both integers and floating-point numbers using IEEE 754 double-precision 64-bit format.

Mechanical Rules:
- All numbers are floating-point (no separate integer type until BigInt)
- Range: -(2^53 - 1) to (2^53 - 1) for safe integers
- Special numeric values: Infinity, -Infinity, NaN (Not-a-Number)
- NaN is the only value in JavaScript that is not equal to itself
- Number.isNaN() checks for actual NaN (not coercion like global isNaN())

Syntax:
let integer = 42;
let floatingPoint = 3.14159;
let scientific = 2.998e8; // 299800000
let hexadecimal = 0xFF; // 255
let binary = 0b1010; // 10
let octal = 0o755; // 493

Example - Number Operations and Edge Cases:
// Basic arithmetic
let sum = 10 + 5; // 15
let difference = 10 - 5; // 5
let product = 10 * 5; // 50
let quotient = 10 / 5; // 2
let remainder = 10 % 3; // 1
let exponentiation = 2 ** 3; // 8

// Floating-point precision issues
console.log(0.1 + 0.2); // 0.30000000000000004 (not exactly 0.3!)
console.log(0.1 + 0.2 === 0.3); // false

// Safe comparison with epsilon
const EPSILON = Number.EPSILON; // Smallest difference between two representable numbers
function areFloatsEqual(a, b) {
    return Math.abs(a - b) < EPSILON;
}
console.log(areFloatsEqual(0.1 + 0.2, 0.3)); // true

// Special numeric values
console.log(1 / 0); // Infinity
console.log(-1 / 0); // -Infinity
console.log(0 / 0); // NaN
console.log(Math.sqrt(-1)); // NaN

// NaN behavior
console.log(NaN === NaN); // false (NaN is not equal to itself!)
console.log(Number.isNaN(NaN)); // true
console.log(Number.isNaN("hello")); // false (does NOT coerce)
console.log(isNaN("hello")); // true (global isNaN coerces to number first)

// Safe integer boundaries
console.log(Number.MAX_SAFE_INTEGER); // 9007199254740991
console.log(Number.MIN_SAFE_INTEGER); // -9007199254740991
console.log(Number.isSafeInteger(9007199254740991)); // true
console.log(Number.isSafeInteger(9007199254740992)); // false


TYPE 5: BIGINT
The BigInt type represents integers of arbitrary precision, beyond the safe integer limit of Number.

Mechanical Rules:
- Created by appending 'n' to integer literal or using BigInt() constructor
- Cannot mix BigInt and Number in arithmetic operations without explicit conversion
- Cannot use Math object methods with BigInt
- Comparison operators work between BigInt and Number
- Division truncates toward zero (no fractional part)

Syntax:
let bigInteger = 9007199254740991n;
let constructedBigInt = BigInt("9007199254740991");

Example - BigInt Operations:
// Creating BigInt values
let largeNumber = 123456789012345678901234567890n;
let fromString = BigInt("999999999999999999999999");

// Arithmetic operations
let bigSum = 100n + 50n; // 150n
let bigProduct = 10n * 20n; // 200n
let bigDivision = 10n / 3n; // 3n (truncates, no decimal)
let bigRemainder = 10n % 3n; // 1n
let bigExponent = 2n ** 100n; // Very large number

// Cannot mix types
try {
    let invalid = 10n + 5; // TypeError
} catch (error) {
    console.log("Cannot mix BigInt and Number");
}

// Explicit conversion required
let mixed = 10n + BigInt(5); // 15n
let mixedToNumber = Number(10n) + 5; // 15

// Comparison works across types
console.log(10n === 10); // false (different types)
console.log(10n == 10); // true (loose equality coerces)
console.log(10n < 20); // true
console.log(5n > 3); // true

// Use case: precise large integer calculations
function factorial(n) {
    if (n === 0n) return 1n;
    let result = 1n;
    for (let i = 1n; i <= n; i++) {
        result *= i;
    }
    return result;
}
console.log(factorial(50n)); // Exact result without precision loss


TYPE 6: STRING
The string type represents sequences of UTF-16 code units. Strings are immutable.

Mechanical Rules:
- Strings can be delimited by single quotes '', double quotes "", or backticks ``
- Backticks enable template literals with interpolation ${expression}
- Strings are indexed from 0, but individual characters cannot be reassigned
- String methods return new strings (original is never mutated)
- Escape sequences: \n (newline), \t (tab), \\ (backslash), \' \", \u{codepoint}

Syntax:
let singleQuote = 'Hello';
let doubleQuote = "World";
let templateLiteral = `Hello ${doubleQuote}`;

Example - String Operations and Immutability:
// String creation
let greeting = "Hello";
let name = 'Alice';
let message = `${greeting}, ${name}!`; // "Hello, Alice!"

// Immutability demonstration
let original = "immutable";
original[0] = "I"; // This does NOTHING (no error in non-strict mode)
console.log(original); // Still "immutable"

// String methods (all return new strings)
let text = "JavaScript";
console.log(text.toUpperCase()); // "JAVASCRIPT"
console.log(text.toLowerCase()); // "javascript"
console.log(text.slice(0, 4)); // "Java"
console.log(text.substring(4, 10)); // "Script"
console.log(text.charAt(0)); // "J"
console.log(text.indexOf("Script")); // 4
console.log(text.includes("Script")); // true
console.log(text.startsWith("Java")); // true
console.log(text.endsWith("Script")); // true

// String concatenation
let part1 = "Hello";
let part2 = "World";
let combined = part1 + " " + part2; // "Hello World"
let templated = `${part1} ${part2}`; // "Hello World" (preferred)

// Multi-line strings
let multiLine = `This is line 1
This is line 2
This is line 3`;

let withEscapes = "Line 1\nLine 2\nLine 3";

// Escape sequences
let escaped = "He said, \"Hello!\""; // He said, "Hello!"
let path = "C:\\Users\\Documents"; // C:\Users\Documents
let unicode = "\u{1F600}"; // 😀 (emoji via Unicode code point)

// String iteration
let word = "abc";
for (let char of word) {
    console.log(char); // "a", "b", "c"
}


TYPE 7: SYMBOL
The symbol type represents unique, immutable identifiers. Symbols are guaranteed to be unique even if they have the same description.

Mechanical Rules:
- Created via Symbol() function (NOT a constructor, no 'new' keyword)
- Each Symbol() call creates a unique symbol
- Symbol.for(key) creates/retrieves global symbols from a registry
- Symbols are often used as object property keys to avoid name collisions
- Symbol properties are not enumerated in for...in loops or Object.keys()

Syntax:
let uniqueId = Symbol();
let namedSymbol = Symbol("description");
let globalSymbol = Symbol.for("global.key");

Example - Symbol Uniqueness and Usage:
// Creating symbols
let sym1 = Symbol("id");
let sym2 = Symbol("id");
console.log(sym1 === sym2); // false (each Symbol() call is unique)

// Global symbol registry
let globalSym1 = Symbol.for("app.id");
let globalSym2 = Symbol.for("app.id");
console.log(globalSym1 === globalSym2); // true (same key retrieves same symbol)

// Retrieving the key from a global symbol
console.log(Symbol.keyFor(globalSym1)); // "app.id"
console.log(Symbol.keyFor(sym1)); // undefined (not in global registry)

// Using symbols as object property keys
let user = {
    name: "Alice",
    age: 30
};

let idSymbol = Symbol("id");
user[idSymbol] = 12345; // Symbol as property key

console.log(user[idSymbol]); // 12345
console.log(user.name); // "Alice"

// Symbols are hidden from normal enumeration
console.log(Object.keys(user)); // ["name", "age"] (symbol property not included)
for (let key in user) {
    console.log(key); // "name", "age" (symbol property not iterated)
}

// Accessing symbol properties
console.log(Object.getOwnPropertySymbols(user)); // [Symbol(id)]
console.log(Reflect.ownKeys(user)); // ["name", "age", Symbol(id)]

// Use case: preventing property name collisions
let library1 = {
    [Symbol.for("lib.version")]: "1.0.0"
};

let library2 = {
    [Symbol.for("lib.version")]: "2.0.0"
};

// Both libraries can use the same conceptual key without collision
console.log(library1[Symbol.for("lib.version")]); // "1.0.0"
console.log(library2[Symbol.for("lib.version")]); // "2.0.0"


1.2 STRUCTURAL TYPES: OBJECTS AND FUNCTIONS
--------------------------------------------------------------------------------

Beyond primitives, JavaScript has structural types: objects and functions. These are mutable and stored by reference.

OBJECTS
Objects are collections of key-value pairs. Keys are strings or symbols; values can be any type.

Mechanical Rules:
- Objects are created via object literals {} or new Object()
- Properties accessed via dot notation obj.key or bracket notation obj["key"]
- Bracket notation required for dynamic keys or keys with special characters
- Objects are passed by reference (mutations affect all references)

Syntax:
let obj = { key: "value" };
let dynamicKey = "propertyName";
let computed = { [dynamicKey]: "value" };

Example - Object Creation and Manipulation:
// Object literal
let person = {
    firstName: "Alice",
    lastName: "Smith",
    age: 30,
    "full-name": "Alice Smith" // Key with hyphen requires quotes
};

// Property access
console.log(person.firstName); // "Alice"
console.log(person["lastName"]); // "Smith"
console.log(person["full-name"]); // "Alice Smith" (dot notation would fail)

// Dynamic property access
let key = "age";
console.log(person[key]); // 30

// Adding properties
person.email = "alice@example.com";
person["phone"] = "555-1234";

// Computed property names
let propName = "dynamicProp";
let obj = {
    [propName]: "value",
    ["computed" + "Key"]: 123
};
console.log(obj.dynamicProp); // "value"
console.log(obj.computedKey); // 123

// Reference behavior
let original = { value: 10 };
let reference = original; // Both point to same object
reference.value = 20;
console.log(original.value); // 20 (mutation visible through original)

// Shallow copy (creates new object, but nested objects still referenced)
let shallowCopy = { ...original };
shallowCopy.value = 30;
console.log(original.value); // 20 (original unaffected)


FUNCTIONS
Functions are first-class objects in JavaScript. They can be assigned to variables, passed as arguments, and returned from other functions.

Mechanical Rules:
- Functions are defined via function declarations, function expressions, or arrow functions
- Function declarations are hoisted (can be called before definition in code)
- Function expressions are NOT hoisted
- Arrow functions have lexical 'this' binding (inherit from enclosing scope)
- Functions without explicit return statement return undefined

Syntax:
function declaration() { }
let expression = function() { };
let arrow = () => { };

Example - Function Types and Behavior:
// Function declaration (hoisted)
console.log(declaredFunction()); // Works! (hoisted)
function declaredFunction() {
    return "I am hoisted";
}

// Function expression (NOT hoisted)
try {
    console.log(expressionFunction()); // ReferenceError
} catch (error) {
    console.log("Function expression not hoisted");
}
let expressionFunction = function() {
    return "I am not hoisted";
};

// Arrow function (concise syntax)
let add = (a, b) => a + b; // Implicit return
let multiply = (a, b) => {
    return a * b; // Explicit return required with block body
};

console.log(add(2, 3)); // 5
console.log(multiply(2, 3)); // 6

// Functions as first-class citizens
function executeOperation(operation, a, b) {
    return operation(a, b);
}

let result = executeOperation(add, 10, 5); // 15
let result2 = executeOperation((x, y) => x - y, 10, 5); // 5

// Returning functions
function createMultiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

let double = createMultiplier(2);
let triple = createMultiplier(3);
console.log(double(5)); // 10
console.log(triple(5)); // 15


1.3 TYPE CHECKING AND COERCION
--------------------------------------------------------------------------------

JavaScript's dynamic typing system performs automatic type coercion in many contexts. Understanding these rules prevents subtle bugs.

THE TYPEOF OPERATOR
Returns a string indicating the type of the operand.

Syntax:
typeof operand

Example - typeof Behavior and Quirks:
console.log(typeof undefined); // "undefined"
console.log(typeof null); // "object" (historical bug)
console.log(typeof true); // "boolean"
console.log(typeof 42); // "number"
console.log(typeof 42n); // "bigint"
console.log(typeof "text"); // "string"
console.log(typeof Symbol()); // "symbol"
console.log(typeof {}); // "object"
console.log(typeof []); // "object" (arrays are objects)
console.log(typeof function() {}); // "function"

// Checking for null requires explicit comparison
let value = null;
if (typeof value === "object" && value === null) {
    console.log("Value is null");
}

// Checking for arrays
console.log(Array.isArray([])); // true
console.log(Array.isArray({})); // false


IMPLICIT TYPE COERCION
JavaScript automatically converts types in certain operations.

Coercion Rules:
- String concatenation: non-strings converted to strings
- Arithmetic operations: non-numbers converted to numbers
- Logical operations: values converted to booleans
- Comparison operators: complex coercion rules apply

Example - String Coercion:
// The + operator with strings triggers string coercion
console.log("5" + 3); // "53" (number 3 becomes string "3")
console.log("Hello" + true); // "Hellotrue"
console.log("Value: " + null); // "Value: null"
console.log("Result: " + undefined); // "Result: undefined"
console.log("Array: " + [1, 2, 3]); // "Array: 1,2,3"
console.log("Object: " + {}); // "Object: [object Object]"

// Template literals also coerce to string
let num = 42;
console.log(`The answer is ${num}`); // "The answer is 42"


Example - Numeric Coercion:
// Arithmetic operators (except +) trigger numeric coercion
console.log("10" - 5); // 5 (string "10" becomes number 10)
console.log("10" * "2"); // 20
console.log("10" / "2"); // 5
console.log("10" % "3"); // 1
console.log("5" - true); // 4 (true becomes 1)
console.log("5" - false); // 5 (false becomes 0)
console.log("5" - null); // 5 (null becomes 0)
console.log("5" - undefined); // NaN (undefined becomes NaN)

// Unary + operator coerces to number
console.log(+"42"); // 42
console.log(+true); // 1
console.log(+false); // 0
console.log(+null); // 0
console.log(+undefined); // NaN
console.log(+""); // 0
console.log(+"hello"); // NaN

// Explicit conversion
console.log(Number("42")); // 42
console.log(Number("42.5")); // 42.5
console.log(Number("42px")); // NaN
console.log(parseInt("42px")); // 42 (parses until non-digit)
console.log(parseFloat("42.5px")); // 42.5


Example - Boolean Coercion:
// Falsy values: false, 0, -0, 0n, "", null, undefined, NaN
// Everything else is truthy

if ("") {
    console.log("Empty string is truthy");
} else {
    console.log("Empty string is falsy"); // This executes
}

if ("0") {
    console.log("String '0' is truthy"); // This executes (non-empty string)
}

if (0) {
    console.log("Number 0 is truthy");
} else {
    console.log("Number 0 is falsy"); // This executes
}

// Logical operators with coercion
console.log(true && "hello"); // "hello" (returns second operand if first is truthy)
console.log(false && "hello"); // false (returns first operand if falsy)
console.log(null || "default"); // "default" (returns first truthy value)
console.log("value" || "default"); // "value"


EQUALITY COMPARISONS
JavaScript has two equality operators with different coercion behavior.

Loose Equality (==): Performs type coercion before comparison
Strict Equality (===): No type coercion, types must match

Example - Equality Operator Behavior:
// Strict equality (===) - no coercion
console.log(5 === 5); // true
console.log(5 === "5"); // false (different types)
console.log(true === 1); // false
console.log(null === undefined); // false
console.log(NaN === NaN); // false (NaN special case)

// Loose equality (==) - with coercion
console.log(5 == "5"); // true (string "5" coerced to number 5)
console.log(true == 1); // true (true coerced to 1)
console.log(false == 0); // true
console.log(null == undefined); // true (special rule)
console.log("" == 0); // true (empty string coerced to 0)
console.log("0" == 0); // true
console.log(false == ""); // true
console.log(false == "0"); // true

// Unexpected coercion results
console.log("" == "0"); // false (both strings, no coercion)
console.log(0 == ""); // true
console.log(0 == "0"); // true

// Best practice: use strict equality
function safeEquals(a, b) {
    return a === b;
}

// Special case: checking for null or undefined
let value = null;
if (value == null) {
    // This catches BOTH null and undefined
    console.log("Value is nullish");
}

// Equivalent strict check
if (value === null || value === undefined) {
    console.log("Value is nullish");
}


OBJECT COERCION
Objects are coerced to primitives via valueOf() and toString() methods.

Example - Object to Primitive Coercion:
let obj = {
    value: 42,
    valueOf() {
        return this.value;
    },
    toString() {
        return `Object with value ${this.value}`;
    }
};

// Numeric context prefers valueOf()
console.log(obj + 10); // 52 (valueOf returns 42)
console.log(obj * 2); // 84

// String context prefers toString()
console.log(String(obj)); // "Object with value 42"
console.log(`Result: ${obj}`); // "Result: Object with value 42"

// Default object coercion
let plainObj = { a: 1 };
console.log(String(plainObj)); // "[object Object]"
console.log(plainObj + ""); // "[object Object]"


================================================================================
SECTION 2: VARIABLE DECLARATIONS - VAR, LET, CONST
================================================================================

JavaScript provides three keywords for variable declaration, each with distinct scoping and hoisting behavior.

2.1 VAR - FUNCTION-SCOPED VARIABLES
--------------------------------------------------------------------------------

The var keyword declares function-scoped or globally-scoped variables. It exhibits hoisting behavior.

Mechanical Rules:
- Variables declared with var are hoisted to the top of their function scope
- Hoisted declarations are initialized with undefined
- var allows redeclaration in the same scope without error
- var ignores block scope (if, for, while blocks)
- Global var declarations create properties on the global object (window in browsers)

Syntax:
var variableName = value;

Example - var Hoisting Behavior:
// Hoisting demonstration
console.log(hoistedVar); // undefined (declaration hoisted, not assignment)
var hoistedVar = "I am hoisted";
console.log(hoistedVar); // "I am hoisted"

// Equivalent to:
// var hoistedVar; // Declaration hoisted to top
// console.log(hoistedVar); // undefined
// hoistedVar = "I am hoisted"; // Assignment stays in place

// Function scope (not block scope)
function demonstrateVarScope() {
    if (true) {
        var functionScoped = "visible outside block";
    }
    console.log(functionScoped); // "visible outside block" (var ignores block)
}
demonstrateVarScope();

// Redeclaration allowed
var x = 10;
var x = 20; // No error
console.log(x); // 20

// Loop variable leakage
for (var i = 0; i < 3; i++) {
    // Loop body
}
console.log(i); // 3 (i leaked outside loop)


Example - var in Closures (Common Pitfall):
// Problem: var in loop with closures
var functions = [];
for (var i = 0; i < 3; i++) {
    functions.push(function() {
        console.log(i); // All closures reference the same 'i'
    });
}

functions[0](); // 3 (not 0!)
functions[1](); // 3 (not 1!)
functions[2](); // 3 (not 2!)
// All functions see i = 3 because var is function-scoped

// Solution 1: IIFE to create new scope
var functionsFixed = [];
for (var i = 0; i < 3; i++) {
    (function(capturedValue) {
        functionsFixed.push(function() {
            console.log(capturedValue);
        });
    })(i);
}

functionsFixed[0](); // 0
functionsFixed[1](); // 1
functionsFixed[2](); // 2


2.2 LET - BLOCK-SCOPED VARIABLES
--------------------------------------------------------------------------------

The let keyword declares block-scoped variables. It does not allow redeclaration and has a temporal dead zone.

Mechanical Rules:
- Variables declared with let are scoped to the nearest enclosing block {}
- let declarations are hoisted but NOT initialized (temporal dead zone)
- Accessing let variables before declaration throws ReferenceError
- let does not allow redeclaration in the same scope
- let does not create properties on the global object

Syntax:
let variableName = value;

Example - let Block Scope:
// Block scope demonstration
if (true) {
    let blockScoped = "only visible in this block";
    console.log(blockScoped); // "only visible in this block"
}

try {
    console.log(blockScoped); // ReferenceError
} catch (error) {
    console.log("blockScoped not accessible outside block");
}

// Loop variable containment
for (let j = 0; j < 3; j++) {
    // Loop body
}

try {
    console.log(j); // ReferenceError (j not leaked)
} catch (error) {
    console.log("Loop variable j is block-scoped");
}


Example - Temporal Dead Zone (TDZ):
// Temporal Dead Zone: period between entering scope and declaration
try {
    console.log(tdzVariable); // ReferenceError (in TDZ)
} catch (error) {
    console.log("Cannot access tdzVariable before declaration");
}

let tdzVariable = "now initialized";
console.log(tdzVariable); // "now initialized"

// TDZ applies even with typeof
try {
    typeof undeclaredLet; // ReferenceError if let undeclaredLet exists below
} catch (error) {
    console.log("typeof throws error in TDZ");
}

// Redeclaration not allowed
let y = 10;
try {
    let y = 20; // SyntaxError in same scope
} catch (error) {
    console.log("Cannot redeclare let variable");
}


Example - let in Closures (Fixes var Problem):
// let creates new binding for each iteration
let functionsWithLet = [];
for (let i = 0; i < 3; i++) {
    functionsWithLet.push(function() {
        console.log(i); // Each closure captures its own 'i'
    });
}

functionsWithLet[0](); // 0
functionsWithLet[1](); // 1
functionsWithLet[2](); // 2


2.3 CONST - BLOCK-SCOPED CONSTANTS
--------------------------------------------------------------------------------

The const keyword declares block-scoped constants. The binding is immutable, but object/array contents can be mutated.

Mechanical Rules:
- const variables must be initialized at declaration
- const binding cannot be reassigned
- const has same block scope and TDZ behavior as let
- For objects/arrays, the reference is constant but contents are mutable
- const does not make values immutable (only the binding)

Syntax:
const constantName = value;

Example - const Immutable Binding:
// Must initialize at declaration
try {
    const uninitialized; // SyntaxError
} catch (error) {
    console.log("const must be initialized");
}

const PI = 3.14159;
console.log(PI); // 3.14159

// Cannot reassign
try {
    PI = 3.14; // TypeError
} catch (error) {
    console.log("Cannot reassign const variable");
}


Example - const with Objects and Arrays (Mutable Contents):
// const with object: reference is constant, properties are mutable
const person = {
    name: "Alice",
    age: 30
};

person.age = 31; // Allowed (mutating property)
person.email = "alice@example.com"; // Allowed (adding property)
console.log(person); // { name: "Alice", age: 31, email: "alice@example.com" }

try {
    person = { name: "Bob" }; // TypeError (cannot reassign binding)
} catch (error) {
    console.log("Cannot reassign const object");
}

// const with array: reference is constant, elements are mutable
const numbers = [1, 2, 3];
numbers.push(4); // Allowed (mutating array)
numbers[0] = 10; // Allowed (modifying element)
console.log(numbers); // [10, 2, 3, 4]

try {
    numbers = [5, 6, 7]; // TypeError (cannot reassign binding)
} catch (error) {
    console.log("Cannot reassign const array");
}

// Creating truly immutable objects
const immutablePerson = Object.freeze({
    name: "Alice",
    age: 30
});

immutablePerson.age = 31; // Silently fails in non-strict mode
console.log(immutablePerson.age); // 30 (unchanged)

// Deep freeze for nested objects
function deepFreeze(obj) {
    Object.freeze(obj);
    Object.values(obj).forEach(value => {
        if (typeof value === "object" && value !== null) {
            deepFreeze(value);
        }
    });
    return obj;
}

const deepImmutable = deepFreeze({
    user: {
        name: "Alice",
        settings: {
            theme: "dark"
        }
    }
});

deepImmutable.user.settings.theme = "light"; // Silently fails
console.log(deepImmutable.user.settings.theme); // "dark" (unchanged)

Choosing the Right Declaration:
1. Default to const for values that won't be reassigned
2. Use let when reassignment is necessary
3. Avoid var in modern JavaScript (legacy compatibility only)

Example - Idiomatic Variable Usage:

const API_ENDPOINT = "https://api.example.com";

    counter += i;
}

    status = "complete";
}

// Avoid: var in modern code
// var legacyVariable = "avoid this";

JavaScript provides a rich set of operators for manipulating values. Understanding operator precedence and associativity is critical for correct expression evaluation.

3.1 ARITHMETIC OPERATORS
--------------------------------------------------------------------------------

Arithmetic operators perform mathematical calculations on numeric operands.

Operator List:
+ (addition/concatenation)
- (subtraction)
* (multiplication)
/ (division)
% (remainder/modulo)
** (exponentiation)
++ (increment)
-- (decrement)

Example - Basic Arithmetic:
let a = 10;

console.log(a + b); // 13
console.log(a - b); // 7
console.log(a * b); // 30
console.log(a / b); // 3.3333333333333335
console.log(a % b); // 1 (remainder)
console.log(a ** b); // 1000 (10^3)

// Division by zero
console.log(10 / 0); // Infinity
console.log(-10 / 0); // -Infinity
console.log(0 / 0); // NaN

// Modulo with negative numbers
console.log(10 % 3); // 1
console.log(-10 % 3); // -1 (sign follows dividend)
console.log(10 % -3); // 1


Example - Increment and Decrement:
// Prefix increment: increments then returns value
let x = 5;
console.log(++x); // 6 (x is now 6, returns 6)
console.log(x); // 6

// Postfix increment: returns value then increments
let y = 5;
console.log(y++); // 5 (returns 5, then y becomes 6)
console.log(y); // 6

// Prefix decrement
let m = 5;
console.log(--m); // 4 (m is now 4, returns 4)

// Postfix decrement
let n = 5;
console.log(n--); // 5 (returns 5, then n becomes 4)
console.log(n); // 4

// In expressions
let counter = 10;
let result = counter++ * 2; // Uses 10, then increments
console.log(result); // 20
console.log(counter); // 11

let counter2 = 10;
let result2 = ++counter2 * 2; // Increments to 11, then uses 11
console.log(result2); // 22
console.log(counter2); // 11


Example - Unary Plus and Minus:

console.log(+"42"); // 42
console.log(+true); // 1
console.log(+false); // 0
console.log(+""); // 0
console.log(+"hello"); // NaN

// Unary minus negates and converts to number
console.log(-"42"); // -42
console.log(-true); // -1
console.log(-false); // -0

// Practical use: converting string to number
let stringNumber = "123";
let numericValue = +stringNumber;
console.log(typeof numericValue); // "number"
console.log(numericValue); // 123

Example - Comparison Operations:

// Loose equality (with coercion)

3.4 LOGICAL OPERATORS
--------------------------------------------------------------------------------

Logical operators perform boolean logic and short-circuit evaluation.

Operator List:
&& (logical AND)
|| (logical OR)
! (logical NOT)
?? (nullish coalescing)

Example - Logical AND (&&):
// Returns first falsy value or last value if all truthy
console.log(true && true); // true
console.log(true && false); // false
console.log(false && true); // false

console.log(5 && 10); // 10 (both truthy, returns last)
console.log(0 && 10); // 0 (first is falsy, returns it)
console.log(null && "value"); // null (first is falsy)
console.log("hello" && "world"); // "world" (both truthy, returns last)

// Short-circuit: second operand not evaluated if first is falsy
let sideEffect = false;
false && (sideEffect = true);
console.log(sideEffect); // false (assignment never executed)


Example - Logical OR (||):
// Returns first truthy value or last value if all falsy
console.log(true || false); // true
console.log(false || true); // true
console.log(false || false); // false

console.log(5 || 10); // 5 (first is truthy, returns it)
console.log(0 || 10); // 10 (first is falsy, continues)
console.log(null || "default"); // "default"
console.log("" || "fallback"); // "fallback"
console.log(0 || false || null || "value"); // "value"

// Short-circuit: second operand not evaluated if first is truthy
let effect = false;
true || (effect = true);
console.log(effect); // false (assignment never executed)

// Common pattern: default values
function greet(name) {
    name = name || "Guest"; // If name is falsy, use "Guest"
    console.log(`Hello, ${name}`);
}
greet("Alice"); // "Hello, Alice"
greet(); // "Hello, Guest"


Example - Logical NOT (!):
// Inverts boolean value
console.log(!true); // false
console.log(!false); // true

// Coerces to boolean then inverts
console.log(!0); // true (0 is falsy)
console.log(!1); // false (1 is truthy)
console.log(!""); // true (empty string is falsy)
console.log(!"hello"); // false (non-empty string is truthy)

// Double negation converts to boolean
console.log(!!"value"); // true
console.log(!!0); // false
console.log(!!null); // false


Example - Nullish Coalescing (??):
// Returns right operand only if left is null or undefined
// Unlike ||, it does NOT treat 0, false, "" as nullish

console.log(null ?? "default"); // "default"
console.log(undefined ?? "default"); // "default"
console.log(0 ?? "default"); // 0 (0 is not nullish)
console.log(false ?? "default"); // false (false is not nullish)
console.log("" ?? "default"); // "" (empty string is not nullish)

// Comparison with ||
console.log(0 || "fallback"); // "fallback" (0 is falsy)
console.log(0 ?? "fallback"); // 0 (0 is not nullish)

console.log("" || "fallback"); // "fallback" (empty string is falsy)
console.log("" ?? "fallback"); // "" (empty string is not nullish)

// Use case: distinguishing between null/undefined and other falsy values
function processValue(value) {
    let processed = value ?? "no value provided";
    console.log(processed);
}

processValue(null); // "no value provided"
processValue(undefined); // "no value provided"
processValue(0); // 0
processValue(false); // false
processValue(""); // ""


3.5 BITWISE OPERATORS
--------------------------------------------------------------------------------

Bitwise operators treat operands as 32-bit signed integers and perform bit-level operations.

Operator List:
& (AND)
| (OR)
^ (XOR)
~ (NOT)
<< (left shift)
>> (sign-propagating right shift)
>>> (zero-fill right shift)

Example - Bitwise AND, OR, XOR:
// Bitwise AND: 1 if both bits are 1
console.log(5 & 3); // 1
// 5: 0101
// 3: 0011
// &: 0001 = 1

// Bitwise OR: 1 if either bit is 1
console.log(5 | 3); // 7
// 5: 0101
// 3: 0011
// |: 0111 = 7

// Bitwise XOR: 1 if bits are different
console.log(5 ^ 3); // 6
// 5: 0101
// 3: 0011
// ^: 0110 = 6

// Bitwise NOT: inverts all bits
console.log(~5); // -6
// 5: 00000000000000000000000000000101
// ~: 11111111111111111111111111111010 = -6 (two's complement)


Example - Bit Shifts:
// Left shift: shifts bits left, fills with zeros
console.log(5 << 1); // 10
// 5: 0101
// <<1: 1010 = 10 (equivalent to multiplying by 2)

console.log(5 << 2); // 20 (equivalent to multiplying by 4)

// Sign-propagating right shift: shifts right, preserves sign bit
console.log(20 >> 1); // 10
// 20: 10100
// >>1: 01010 = 10 (equivalent to dividing by 2)

console.log(-20 >> 1); // -10 (sign bit preserved)

// Zero-fill right shift: shifts right, fills with zeros
console.log(20 >>> 1); // 10
console.log(-20 >>> 1); // 2147483638 (treats as unsigned)


Example - Practical Bitwise Use Cases:
// Checking if number is even or odd
function isEven(n) {
    return (n & 1) === 0; // Last bit is 0 for even numbers
}
console.log(isEven(4)); // true
console.log(isEven(7)); // false

// Swapping values without temporary variable
let p = 5;
let q = 10;
p = p ^ q;
q = p ^ q;
p = p ^ q;
console.log(p, q); // 10, 5

// Setting, clearing, toggling bits (flags)
let flags = 0;
const FLAG_A = 1 << 0; // 0001
const FLAG_B = 1 << 1; // 0010
const FLAG_C = 1 << 2; // 0100

// Set flag
flags |= FLAG_A; // Set FLAG_A
console.log(flags); // 1

// Check flag
console.log((flags & FLAG_A) !== 0); // true (FLAG_A is set)
console.log((flags & FLAG_B) !== 0); // false (FLAG_B is not set)

// Clear flag
flags &= ~FLAG_A; // Clear FLAG_A
console.log(flags); // 0

// Toggle flag
flags ^= FLAG_B; // Toggle FLAG_B
console.log(flags); // 2

3.7 OPERATOR PRECEDENCE AND ASSOCIATIVITY
--------------------------------------------------------------------------------

Operators are evaluated in order of precedence. Operators with the same precedence are evaluated based on associativity (left-to-right or right-to-left).

Precedence Table (highest to lowest):
1. Grouping: ()
2. Member access: . []
3. Function call: ()
4. Postfix increment/decrement: ++ --
5. Prefix increment/decrement, unary: ++ -- ! ~ + - typeof void delete
6. Exponentiation: ** (right-associative)
7. Multiplication/division: * / %
8. Addition/subtraction: + -
9. Bitwise shifts: << >> >>>
10. Relational: < <= > >= in instanceof
11. Equality: == != === !==
12. Bitwise AND: &
13. Bitwise XOR: ^
14. Bitwise OR: |
15. Logical AND: &&
16. Logical OR: ||
17. Nullish coalescing: ??
18. Conditional: ? :
19. Assignment: = += -= *= /= %= **= (right-associative)
20. Comma: ,

Example - Precedence Demonstration:
// Multiplication before addition
console.log(2 + 3 * 4); // 14 (not 20)
// Equivalent to: 2 + (3 * 4)

// Grouping overrides precedence
console.log((2 + 3) * 4); // 20

// Exponentiation is right-associative
console.log(2 ** 3 ** 2); // 512 (not 64)
// Equivalent to: 2 ** (3 ** 2) = 2 ** 9 = 512

// Comparison before logical AND
console.log(5 > 3 && 10 < 20); // true
// Equivalent to: (5 > 3) && (10 < 20)

// Logical AND before logical OR
console.log(false || true && false); // false
// Equivalent to: false || (true && false)

// Assignment is right-associative
let a, b, c;
a = b = c = 10;
// Equivalent to: a = (b = (c = 10))
console.log(a, b, c); // 10 10 10


Example - Complex Precedence Scenarios:
// Multiple operators
let result1 = 10 + 5 * 2 - 3 / 3;
// Step 1: 5 * 2 = 10
// Step 2: 3 / 3 = 1
// Step 3: 10 + 10 = 20
// Step 4: 20 - 1 = 19
console.log(result1); // 19

// Mixing logical and comparison
let result2 = 5 > 3 && 10 < 20 || false;
// Step 1: 5 > 3 = true
// Step 2: 10 < 20 = true
// Step 3: true && true = true
// Step 4: true || false = true
console.log(result2); // true

// Unary operators
let result3 = -5 + 3 * 2;
// Step 1: -5 (unary minus)
// Step 2: 3 * 2 = 6
// Step 3: -5 + 6 = 1
console.log(result3); // 1

// Ternary with logical operators
let result4 = true ? 10 : 20 + 5;
// Ternary has lower precedence than addition
// Step 1: 20 + 5 = 25
// Step 2: true ? 10 : 25 = 10
console.log(result4); // 10

// Parentheses for clarity (best practice)
let result5 = (10 + 5) * (2 - 3) / 3;
// Step 1: (10 + 5) = 15
// Step 2: (2 - 3) = -1
// Step 3: 15 * -1 = -15
// Step 4: -15 / 3 = -5
console.log(result5); // -5

SECTION 4: EXECUTION CONTEXT AND SCOPE
================================================================================

JavaScript's execution model is based on execution contexts, which determine variable accessibility and the value of 'this'.

4.1 EXECUTION CONTEXT FUNDAMENTALS
--------------------------------------------------------------------------------

An execution context is an abstract environment where JavaScript code is evaluated and executed.

Types of Execution Contexts:
1. Global Execution Context: Created when script first runs
2. Function Execution Context: Created when function is invoked
3. Eval Execution Context: Created inside eval() function (avoid in modern code)

Execution Context Components:
- Variable Environment: Stores var declarations and function declarations
- Lexical Environment: Stores let/const declarations and outer environment reference
- This Binding: Value of 'this' keyword

Example - Execution Context Creation:
// Global execution context created
let globalVar = "I am global";

function outerFunction() {
    // Function execution context created when invoked
    let outerVar = "I am in outer function";
    
    function innerFunction() {
        // Another function execution context created
        let innerVar = "I am in inner function";
        console.log(innerVar); // Accessible
        console.log(outerVar); // Accessible (via scope chain)
        console.log(globalVar); // Accessible (via scope chain)
    }
    
    innerFunction();
}

outerFunction();


4.2 THE CALL STACK
--------------------------------------------------------------------------------

The call stack is a data structure that tracks function invocations. It operates on a Last-In-First-Out (LIFO) principle.

Call Stack Mechanics:
1. When script starts, global execution context is pushed onto stack
2. When function is called, new execution context is pushed onto stack
3. When function returns, its execution context is popped from stack
4. Stack overflow occurs if stack size limit is exceeded (usually from infinite recursion)

Example - Call Stack Visualization:
function first() {
    console.log("Inside first");
    second();
    console.log("Back in first");
}

function second() {
    console.log("Inside second");
    third();
    console.log("Back in second");
}

function third() {
    console.log("Inside third");
}

first();

// Call stack progression:
// 1. [global]
// 2. [global, first]
// 3. [global, first, second]
// 4. [global, first, second, third]
// 5. [global, first, second] (third returns)
// 6. [global, first] (second returns)
// 7. [global] (first returns)

// Output:
// Inside first
// Inside second
// Inside third
// Back in second
// Back in first


Example - Stack Overflow:
function recursiveFunction() {
    recursiveFunction(); // Infinite recursion
}

try {
    recursiveFunction();
} catch (error) {
    console.log(error.name); // "RangeError"
    console.log(error.message); // "Maximum call stack size exceeded"
}


4.3 SCOPE AND SCOPE CHAIN
--------------------------------------------------------------------------------

Scope determines the accessibility of variables. JavaScript uses lexical scoping (scope determined by code structure, not runtime).

Scope Types:
1. Global Scope: Variables accessible everywhere
2. Function Scope: Variables accessible within function (var, function declarations)
3. Block Scope: Variables accessible within block {} (let, const)

Scope Chain: When variable is accessed, JavaScript searches current scope, then outer scopes, up to global scope.

Example - Scope Chain Lookup:
let global = "global";

function outer() {
    let outerVar = "outer";
    
    function inner() {
        let innerVar = "inner";
        
        // Scope chain lookup:
        console.log(innerVar); // Found in inner scope
        console.log(outerVar); // Found in outer scope (via scope chain)
        console.log(global); // Found in global scope (via scope chain)
    }
    
    inner();
    
    try {
        console.log(innerVar); // ReferenceError (not in scope chain)
    } catch (error) {
        console.log("innerVar not accessible in outer");
    }
}

outer();


Example - Lexical Scope (Static Scope):
// Scope is determined by code structure, not call location

let value = "global";

function showValue() {
    console.log(value); // Looks up scope chain from where function is DEFINED
}

function outer() {
    let value = "outer";
    showValue(); // Calls function, but scope chain is based on definition location
}

outer(); // Logs "global" (not "outer")
// showValue's scope chain: showValue -> global
// It does NOT include outer's scope


Example - Block Scope with let and const:
// let and const are block-scoped
if (true) {
    let blockScoped = "only in block";
    const alsoBlockScoped = "also only in block";
    var functionScoped = "leaks out of block";
    
    console.log(blockScoped); // Accessible
}

try {
    console.log(blockScoped); // ReferenceError
} catch (error) {
    console.log("blockScoped not accessible outside block");
}

console.log(functionScoped); // "leaks out of block" (var ignores block scope)

// Block scope in loops
for (let i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100);
}
// Logs: 0, 1, 2 (each iteration has its own 'i')

for (var j = 0; j < 3; j++) {
    setTimeout(() => console.log(j), 100);
}
// Logs: 3, 3, 3 (all closures share same 'j')


4.4 HOISTING
--------------------------------------------------------------------------------

Hoisting is JavaScript's behavior of moving declarations to the top of their scope during compilation phase.

Hoisting Rules:
- var declarations are hoisted and initialized with undefined
- let/const declarations are hoisted but NOT initialized (temporal dead zone)
- Function declarations are fully hoisted (declaration and definition)
- Function expressions are NOT hoisted (treated as variable assignments)

Example - var Hoisting:
console.log(hoistedVar); // undefined (declaration hoisted, not assignment)
var hoistedVar = "value";
console.log(hoistedVar); // "value"

// Equivalent to:
// var hoistedVar; // Hoisted to top
// console.log(hoistedVar); // undefined
// hoistedVar = "value";
// console.log(hoistedVar); // "value"


Example - let/const Hoisting and Temporal Dead Zone:
try {
    console.log(letVar); // ReferenceError (in TDZ)
} catch (error) {
    console.log("Cannot access letVar before declaration");
}

let letVar = "value";
console.log(letVar); // "value"

// const behaves identically
try {
    console.log(constVar); // ReferenceError (in TDZ)
} catch (error) {
    console.log("Cannot access constVar before declaration");
}

const constVar = "value";


Example - Function Declaration Hoisting:
// Function declarations are fully hoisted
console.log(hoistedFunction()); // "I am hoisted" (works!)

function hoistedFunction() {
    return "I am hoisted";
}


Example - Function Expression NOT Hoisted:
try {
    console.log(notHoisted()); // TypeError (notHoisted is undefined)
} catch (error) {
    console.log("Function expression not hoisted");
}

var notHoisted = function() {
    return "I am not hoisted";
};

// With let/const, it's a ReferenceError (TDZ)
try {
    console.log(alsoNotHoisted()); // ReferenceError
} catch (error) {
    console.log("Function expression with let not hoisted");
}

let alsoNotHoisted = function() {
    return "I am not hoisted";
};


Example - Hoisting in Nested Scopes:
var x = "global";

function outer() {
    console.log(x); // undefined (not "global")
    var x = "local";
    console.log(x); // "local"
}

outer();

// Equivalent to:
// var x = "global";
// function outer() {
//     var x; // Hoisted to top of function
//     console.log(x); // undefined
//     x = "local";
//     console.log(x); // "local"
// }


4.5 CLOSURES
--------------------------------------------------------------------------------

A closure is a function that retains access to its lexical scope even when executed outside that scope.

Closure Mechanics:
- Inner function captures variables from outer function's scope
- Captured variables remain accessible even after outer function returns
- Each function invocation creates a new closure with its own captured variables

Example - Basic Closure:
function createCounter() {
    let count = 0; // Private variable
    
    return function() {
        count++; // Accesses outer scope variable
        return count;
    };
}

let counter1 = createCounter();
console.log(counter1()); // 1
console.log(counter1()); // 2
console.log(counter1()); // 3

let counter2 = createCounter(); // New closure, independent count
console.log(counter2()); // 1
console.log(counter2()); // 2

console.log(counter1()); // 4 (counter1's count is separate)


Example - Closure with Multiple Functions:
function createPerson(name) {
    let age = 0; // Private variable
    
    return {
        getName() {
            return name;
        },
        getAge() {
            return age;
        },
        setAge(newAge) {
            if (newAge >= 0) {
                age = newAge;
            }
        },
        haveBirthday() {
            age++;
        }
    };
}

let person = createPerson("Alice");
console.log(person.getName()); // "Alice"
console.log(person.getAge()); // 0

person.setAge(25);
console.log(person.getAge()); // 25

person.haveBirthday();
console.log(person.getAge()); // 26

// Cannot access private variables directly
console.log(person.name); // undefined
console.log(person.age); // undefined


Example - Closure in Loops (Classic Problem):
// Problem: var in loop
var functions = [];
for (var i = 0; i < 3; i++) {
    functions.push(function() {
        console.log(i);
    });
}

functions[0](); // 3 (not 0)
functions[1](); // 3 (not 1)
functions[2](); // 3 (not 2)
// All closures capture the same 'i' variable

// Solution 1: IIFE to create new scope
var functionsFixed1 = [];
for (var i = 0; i < 3; i++) {
    (function(captured) {
        functionsFixed1.push(function() {
            console.log(captured);
        });
    })(i);
}

functionsFixed1[0](); // 0
functionsFixed1[1](); // 1
functionsFixed1[2](); // 2

// Solution 2: let (creates new binding per iteration)
let functionsFixed2 = [];
for (let i = 0; i < 3; i++) {
    functionsFixed2.push(function() {
        console.log(i);
    });
}

functionsFixed2[0](); // 0
functionsFixed2[1](); // 1
functionsFixed2[2](); // 2


Example - Practical Closure Use Case (Module Pattern):
const calculator = (function() {
    // Private variables and functions
    let result = 0;
    
    function log(operation, value) {
        console.log(`${operation}: ${value}, result: ${result}`);
    }
    
    // Public API
    return {
        add(value) {
            result += value;
            log("add", value);
            return this;
        },
        subtract(value) {
            result -= value;
            log("subtract", value);
            return this;
        },
        multiply(value) {
            result *= value;
            log("multiply", value);
            return this;
        },
        getResult() {
            return result;
        },
        reset() {
            result = 0;
            return this;
        }
    };
})();

calculator.add(10).multiply(2).subtract(5);
console.log(calculator.getResult()); // 15
calculator.reset();
console.log(calculator.getResult()); // 0


4.6 THE 'THIS' KEYWORD
--------------------------------------------------------------------------------

The 'this' keyword refers to the object that is executing the current function. Its value is determined by how the function is called, not where it's defined.

'this' Binding Rules (in order of precedence):
1. new binding: 'this' is the newly created object
2. Explicit binding: 'this' is the object passed to call/apply/bind
3. Implicit binding: 'this' is the object before the dot
4. Default binding: 'this' is global object (or undefined in strict mode)
5. Arrow functions: 'this' is lexically inherited from enclosing scope

Example - Default Binding:
function showThis() {
    console.log(this);
}

showThis(); // Window (in browser) or global (in Node.js)

// In strict mode
"use strict";
function showThisStrict() {
    console.log(this);
}

showThisStrict(); // undefined


Example - Implicit Binding:
const person = {
    name: "Alice",
    greet() {
        console.log(`Hello, I am ${this.name}`);
    }
};

person.greet(); // "Hello, I am Alice" (this = person)

// Losing implicit binding
const greetFunction = person.greet;
greetFunction(); // "Hello, I am undefined" (this = global/undefined)


Example - Explicit Binding (call, apply, bind):
function introduce(greeting, punctuation) {
    console.log(`${greeting}, I am ${this.name}${punctuation}`);
}

const user = { name: "Bob" };

// call: invokes function with specified 'this' and individual arguments
introduce.call(user, "Hello", "!"); // "Hello, I am Bob!"

// apply: invokes function with specified 'this' and array of arguments
introduce.apply(user, ["Hi", "."]); // "Hi, I am Bob."

// bind: returns new function with bound 'this' (does not invoke)
const boundIntroduce = introduce.bind(user);
boundIntroduce("Hey", "?"); // "Hey, I am Bob?"

// Partial application with bind
const boundWithGreeting = introduce.bind(user, "Greetings");
boundWithGreeting("!!!"); // "Greetings, I am Bob!!!"


Example - new Binding:
function Person(name, age) {
    this.name = name;
    this.age = age;
    this.greet = function() {
        console.log(`I am ${this.name}, age ${this.age}`);
    };
}

const alice = new Person("Alice", 30);
alice.greet(); // "I am Alice, age 30" (this = alice)

// What 'new' does:
// 1. Creates new empty object
// 2. Sets object's prototype to constructor's prototype
// 3. Binds 'this' to the new object
// 4. Executes constructor function
// 5. Returns the object (unless constructor explicitly returns an object)


Example - Arrow Functions and Lexical 'this':
// Arrow functions do NOT have their own 'this'
// They inherit 'this' from enclosing scope

const team = {
    name: "Engineering",
    members: ["Alice", "Bob", "Charlie"],
    
    // Regular function: 'this' is team
    showMembersRegular() {
        this.members.forEach(function(member) {
            // Problem: 'this' is undefined (or global in non-strict mode)
            console.log(`${member} is in ${this.name}`);
        });
    },
    
    // Solution 1: Arrow function inherits 'this'
    showMembersArrow() {
        this.members.forEach(member => {
            // Arrow function inherits 'this' from showMembersArrow
            console.log(`${member} is in ${this.name}`);
        });
    },
    
    // Solution 2: Bind 'this' explicitly
    showMembersBound() {
        this.members.forEach(function(member) {
            console.log(`${member} is in ${this.name}`);
        }.bind(this));
    },
    
    // Solution 3: Capture 'this' in variable
    showMembersCaptured() {
        const self = this;
        this.members.forEach(function(member) {
            console.log(`${member} is in ${self.name}`);
        });
    }
};

team.showMembersArrow();
// Alice is in Engineering
// Bob is in Engineering
// Charlie is in Engineering


Example - 'this' in Event Handlers:
// In DOM event handlers, 'this' refers to the element
// (This is a conceptual example; actual DOM not available here)

/*
const button = document.querySelector("button");

button.addEventListener("click", function() {
    console.log(this); // The button element
    this.textContent = "Clicked!";
});

// Arrow function does NOT work for this use case
button.addEventListener("click", () => {
    console.log(this); // Window (or enclosing scope's 'this')
    // Cannot access button element via 'this'
});
*/


Example - Complex 'this' Scenarios:
const obj = {
    value: 42,
    getValue() {
        return this.value;
    },
    getValueArrow: () => {
        return this.value; // 'this' is NOT obj (lexically inherited)
    },
    nested: {
        value: 100,
        getValue() {
            return this.value;
        }
    }
};

console.log(obj.getValue()); // 42 (this = obj)
console.log(obj.getValueArrow()); // undefined (this = global/window)
console.log(obj.nested.getValue()); // 100 (this = obj.nested)

// Extracting method loses context
const extracted = obj.getValue;
console.log(extracted()); // undefined (this = global/undefined)

// Binding preserves context
const bound = obj.getValue.bind(obj);
console.log(bound()); // 42 (this = obj)

SECTION 5: STRICT MODE
================================================================================

Strict mode is a restricted variant of JavaScript that eliminates silent errors, improves performance, and prohibits problematic syntax.

5.1 ENABLING STRICT MODE
--------------------------------------------------------------------------------

Strict mode is enabled by placing "use strict"; at the beginning of a script or function.

Syntax:
"use strict"; // At top of file (entire script)

function strictFunction() {
    "use strict"; // Only this function
    // Strict mode code
}

Example - Enabling Strict Mode:
// Global strict mode
"use strict";

let x = 10; // Valid

try {
    y = 20; // ReferenceError (implicit global not allowed)
} catch (error) {
    console.log("Cannot create implicit global in strict mode");
}

// Function-level strict mode
function normalFunction() {
    implicitGlobal = 30; // Creates global variable (bad practice)
}

function strictFunction() {
    "use strict";
    try {
        anotherImplicit = 40; // ReferenceError
    } catch (error) {
        console.log("Strict mode prevents implicit globals");
    }
}

normalFunction();
console.log(implicitGlobal); // 30 (global created)

strictFunction();


5.2 STRICT MODE RESTRICTIONS
--------------------------------------------------------------------------------

Strict mode enforces several restrictions that prevent common errors.

RESTRICTION 1: No Implicit Globals
Variables must be declared before use.

Example:
"use strict";

try {
    undeclaredVariable = 10; // ReferenceError
} catch (error) {
    console.log("Must declare variables");
}

let declaredVariable = 10; // Valid


RESTRICTION 2: No Duplicate Parameter Names
Function parameters must have unique names.

Example:
try {
    eval(`
        "use strict";
        function duplicate(a, a, b) { // SyntaxError
            return a + b;
        }
    `);
} catch (error) {
    console.log("Duplicate parameters not allowed");
}

// Valid in non-strict mode (last 'a' wins)
function duplicateNonStrict(a, a, b) {
    return a + b;
}
console.log(duplicateNonStrict(1, 2, 3)); // 5 (uses second 'a' = 2)


RESTRICTION 3: No Octal Literals
Octal number literals (0-prefixed) are not allowed.

Example:
"use strict";

try {
    eval(`
        "use strict";
        let octal = 0755; // SyntaxError
    `);
} catch (error) {
    console.log("Octal literals not allowed");
}

// Valid: explicit octal with 0o prefix
let validOctal = 0o755; // 493 in decimal
console.log(validOctal); // 493


RESTRICTION 4: Cannot Delete Variables, Functions, or Non-Configurable Properties
The delete operator has stricter rules.

Example:
"use strict";

let variable = 10;
try {
    delete variable; // SyntaxError
} catch (error) {
    console.log("Cannot delete variables");
}

function func() {}
try {
    delete func; // SyntaxError
} catch (error) {
    console.log("Cannot delete functions");
}

let obj = {};
Object.defineProperty(obj, "fixed", {
    value: 42,
    configurable: false
});

try {
    delete obj.fixed; // TypeError
} catch (error) {
    console.log("Cannot delete non-configurable property");
}


RESTRICTION 5: 'this' is Undefined in Functions (Not Global Object)
Default 'this' binding is undefined instead of global object.

Example:
"use strict";

function showThis() {
    console.log(this);
}

showThis(); // undefined (not global object)

// In non-strict mode
function showThisNonStrict() {
    console.log(this);
}
// Would log global object (Window in browser)


RESTRICTION 6: Cannot Use 'with' Statement
The 'with' statement is prohibited.

Example:
"use strict";

let obj = { a: 1, b: 2 };

try {
    eval(`
        "use strict";
        with (obj) { // SyntaxError
            console.log(a);
        }
    `);
} catch (error) {
    console.log("'with' statement not allowed");
}


RESTRICTION 7: Reserved Words Cannot Be Used as Identifiers
Future reserved words cannot be used as variable names.

Example:
"use strict";

try {
    eval(`
        "use strict";
        let implements = 10; // SyntaxError
    `);
} catch (error) {
    console.log("Reserved words cannot be identifiers");
}

// Reserved in strict mode:
// implements, interface, let, package, private, protected, public, static, yield


RESTRICTION 8: Assignment to Read-Only Properties Throws Error
Silent failures become explicit errors.

Example:
"use strict";

let obj = {};
Object.defineProperty(obj, "readOnly", {
    value: 42,
    writable: false
});

try {
    obj.readOnly = 100; // TypeError
} catch (error) {
    console.log("Cannot assign to read-only property");
}

// Assignment to getter-only property
let obj2 = {
    get prop() {
        return 42;
    }
};

try {
    obj2.prop = 100; // TypeError
} catch (error) {
    console.log("Cannot assign to getter-only property");
}


5.3 STRICT MODE BENEFITS
--------------------------------------------------------------------------------

Strict mode provides several advantages for code quality and performance.

BENEFIT 1: Catches Common Mistakes
Prevents accidental global variable creation and other errors.

Example:
"use strict";

function calculateTotal(price, quantity) {
    // Typo: 'totle' instead of 'total'
    try {
        totle = price * quantity; // ReferenceError (caught immediately)
    } catch (error) {
        console.log("Typo caught by strict mode");
    }
    
    let total = price * quantity; // Correct
    return total;
}

calculateTotal(10, 5);


BENEFIT 2: Improves Performance
Eliminates features that make optimization difficult.

Example:
"use strict";

// Strict mode allows engines to optimize better
// - No 'with' statement (unpredictable scope)
// - No implicit globals (predictable variable resolution)
// - No arguments.callee (enables tail call optimization)

function optimizable(a, b) {
    return a + b;
}


BENEFIT 3: Prepares for Future JavaScript Versions
Reserves keywords and prohibits syntax that may conflict with future features.

Example:
"use strict";

// These are reserved for future use
try {
    eval(`
        "use strict";
        let private = 10; // SyntaxError
    `);
} catch (error) {
    console.log("Future reserved word");
}


BENEFIT 4: Safer 'eval' and 'arguments'
Variables introduced by eval do not leak into surrounding scope.

Example:
"use strict";

function testEval() {
    eval("let x = 10");
    try {
        console.log(x); // ReferenceError (x is scoped to eval)
    } catch (error) {
        console.log("eval variables are scoped");
    }
}

testEval();

// In non-strict mode, x would leak into function scope

SECTION 6: MEMORY MODEL AND REFERENCE BEHAVIOR
================================================================================

Understanding how JavaScript stores and manages values is critical for avoiding bugs related to mutation and reference sharing.

6.1 PRIMITIVE VS REFERENCE VALUES
--------------------------------------------------------------------------------

Primitives are stored by value; objects are stored by reference.

Primitive Types (Stored by Value):
- undefined, null, boolean, number, bigint, string, symbol
- Copying a primitive creates an independent copy
- Comparison compares actual values

Reference Types (Stored by Reference):
- Objects, arrays, functions
- Copying a reference creates a new reference to the same object
- Comparison compares references, not contents

Example - Primitive Value Behavior:
let a = 10;
let b = a; // Copy value
b = 20;

console.log(a); // 10 (unchanged)
console.log(b); // 20

// String immutability
let str1 = "hello";
let str2 = str1;
str2 = "world";

console.log(str1); // "hello" (unchanged)
console.log(str2); // "world"


Example - Reference Value Behavior:
let obj1 = { value: 10 };
let obj2 = obj1; // Copy reference (both point to same object)
obj2.value = 20;

console.log(obj1.value); // 20 (mutation visible through obj1)
console.log(obj2.value); // 20

// Array reference behavior
let arr1 = [1, 2, 3];
let arr2 = arr1; // Copy reference
arr2.push(4);

console.log(arr1); // [1, 2, 3, 4] (mutation visible)
console.log(arr2); // [1, 2, 3, 4]

// Function reference behavior
let func1 = function() { return 42; };
let func2 = func1; // Copy reference
console.log(func1 === func2); // true (same reference)


Example - Comparison Behavior:
// Primitive comparison (by value)
console.log(10 === 10); // true
console.log("hello" === "hello"); // true

// Object comparison (by reference)
console.log({} === {}); // false (different objects)
console.log([] === []); // false (different arrays)

let obj = { a: 1 };
let ref = obj;
console.log(obj === ref); // true (same reference)

// Arrays with same contents are not equal
let arr1 = [1, 2, 3];
let arr2 = [1, 2, 3];
console.log(arr1 === arr2); // false (different references)


6.2 COPYING OBJECTS AND ARRAYS
--------------------------------------------------------------------------------

Creating independent copies requires explicit copying strategies.

SHALLOW COPY
Copies top-level properties, but nested objects remain referenced.

Example - Shallow Copy Techniques:
// Object shallow copy with spread operator
let original = { a: 1, b: 2, nested: { c: 3 } };
let shallowCopy = { ...original };

shallowCopy.a = 10;
console.log(original.a); // 1 (top-level property independent)

shallowCopy.nested.c = 30;
console.log(original.nested.c); // 30 (nested object still referenced!)

// Object.assign for shallow copy
let copy2 = Object.assign({}, original);
copy2.b = 20;
console.log(original.b); // 2 (independent)

copy2.nested.c = 300;
console.log(original.nested.c); // 300 (nested still referenced)

// Array shallow copy
let originalArr = [1, 2, [3, 4]];
let shallowArr = [...originalArr];

shallowArr[0] = 10;
console.log(originalArr[0]); // 1 (independent)

shallowArr[2][0] = 30;
console.log(originalArr[2][0]); // 30 (nested array still referenced!)

// Array.slice() for shallow copy
let copy3 = originalArr.slice();
copy3[1] = 20;
console.log(originalArr[1]); // 2 (independent)


DEEP COPY
Recursively copies all nested objects, creating fully independent copies.

Example - Deep Copy Techniques:
// JSON serialization (simple but limited)
let original = {
    a: 1,
    b: { c: 2, d: { e: 3 } },
    arr: [1, 2, { f: 4 }]
};

let deepCopy = JSON.parse(JSON.stringify(original));
deepCopy.b.d.e = 300;
console.log(original.b.d.e); // 3 (independent!)

deepCopy.arr[2].f = 400;
console.log(original.arr[2].f); // 4 (independent!)

// Limitations of JSON method:
let problematic = {
    date: new Date(),
    func: function() {},
    undef: undefined,
    symbol: Symbol("id"),
    circular: null
};
problematic.circular = problematic; // Circular reference

try {
    JSON.parse(JSON.stringify(problematic));
} catch (error) {
    console.log("JSON method fails with circular references");
}

let jsonCopy = JSON.parse(JSON.stringify({
    date: new Date(),
    func: function() {},
    undef: undefined
}));

console.log(jsonCopy.date instanceof Date); // false (becomes string)
console.log(jsonCopy.func); // undefined (functions not serialized)
console.log("undef" in jsonCopy); // false (undefined properties omitted)


// Manual deep copy function
function deepClone(obj, seen = new WeakMap()) {
    // Handle primitives and null
    if (obj === null || typeof obj !== "object") {
        return obj;
    }
    
    // Handle circular references
    if (seen.has(obj)) {
        return seen.get(obj);
    }
    
    // Handle Date
    if (obj instanceof Date) {
        return new Date(obj.getTime());
    }
    
    // Handle Array
    if (Array.isArray(obj)) {
        let arrCopy = [];
        seen.set(obj, arrCopy);
        obj.forEach((item, index) => {
            arrCopy[index] = deepClone(item, seen);
        });
        return arrCopy;
    }
    
    // Handle Object
    let objCopy = {};
    seen.set(obj, objCopy);
    Object.keys(obj).forEach(key => {
        objCopy[key] = deepClone(obj[key], seen);
    });
    
    return objCopy;
}

// Test deep clone
let complex = {
    a: 1,
    b: { c: 2 },
    d: [3, { e: 4 }],
    date: new Date()
};
complex.self = complex; // Circular reference

let cloned = deepClone(complex);
cloned.b.c = 200;
cloned.d[1].e = 400;

console.log(complex.b.c); // 2 (independent)
console.log(complex.d[1].e); // 4 (independent)
console.log(cloned.date instanceof Date); // true (Date preserved)
console.log(cloned.self === cloned); // true (circular reference preserved)


6.3 FUNCTION PARAMETERS AND REFERENCES
--------------------------------------------------------------------------------

Function parameters behave according to the same value/reference rules.

Example - Primitives Passed by Value:
function modifyPrimitive(value) {
    value = 100; // Modifies local copy only
}

let num = 10;
modifyPrimitive(num);
console.log(num); // 10 (unchanged)


Example - Objects Passed by Reference:
function modifyObject(obj) {
    obj.value = 100; // Mutates original object
}

let object = { value: 10 };
modifyObject(object);
console.log(object.value); // 100 (changed!)


function reassignObject(obj) {
    obj = { value: 200 }; // Reassigns local reference only
}

let object2 = { value: 10 };
reassignObject(object2);
console.log(object2.value); // 10 (unchanged, reassignment doesn't affect original)


Example - Arrays Passed by Reference:
function modifyArray(arr) {
    arr.push(4); // Mutates original array
}

let array = [1, 2, 3];
modifyArray(array);
console.log(array); // [1, 2, 3, 4] (changed!)


function reassignArray(arr) {
    arr = [10, 20, 30]; // Reassigns local reference only
}

let array2 = [1, 2, 3];
reassignArray(array2);
console.log(array2); // [1, 2, 3] (unchanged)


Example - Defensive Copying:
// To prevent unintended mutations, copy parameters
function safeModify(obj) {
    let copy = { ...obj }; // Shallow copy
    copy.value = 100;
    return copy;
}

let original = { value: 10 };
let modified = safeModify(original);
console.log(original.value); // 10 (unchanged)
console.log(modified.value); // 100


function safeArrayModify(arr) {
    let copy = [...arr]; // Shallow copy
    copy.push(4);
    return copy;
}

let originalArr = [1, 2, 3];
let modifiedArr = safeArrayModify(originalArr);
console.log(originalArr); // [1, 2, 3] (unchanged)
console.log(modifiedArr); // [1, 2, 3, 4]

SECTION 7: ADVANCED TYPE CONCEPTS
================================================================================

7.1 TYPE CONVERSION ALGORITHMS
--------------------------------------------------------------------------------

JavaScript uses specific algorithms to convert values between types.

ToPrimitive Algorithm:
Converts objects to primitives using valueOf() and toString() methods.

Example - ToPrimitive Conversion:
let obj = {
    valueOf() {
        console.log("valueOf called");
        return 42;
    },
    toString() {
        console.log("toString called");
        return "object";
    }
};

// Numeric context prefers valueOf
console.log(obj + 10); // valueOf called, result: 52
console.log(obj * 2); // valueOf called, result: 84

// String context prefers toString
console.log(String(obj)); // toString called, result: "object"
console.log(`Value: ${obj}`); // toString called, result: "Value: object"


ToNumber Algorithm:
Converts values to numbers.

Conversion Rules:
- undefined → NaN
- null → 0
- true → 1, false → 0
- string → parsed number or NaN
- object → ToPrimitive(hint: number), then ToNumber

Example - ToNumber Conversion:
console.log(Number(undefined)); // NaN
console.log(Number(null)); // 0
console.log(Number(true)); // 1
console.log(Number(false)); // 0
console.log(Number("42")); // 42
console.log(Number("42.5")); // 42.5
console.log(Number("42px")); // NaN
console.log(Number("")); // 0
console.log(Number("   ")); // 0

// Object conversion
let obj = {
    valueOf() { return 100; }
};
console.log(Number(obj)); // 100


ToString Algorithm:
Converts values to strings.

Conversion Rules:
- undefined → "undefined"
- null → "null"
- true → "true", false → "false"
- number → string representation
- object → ToPrimitive(hint: string), then ToString

Example - ToString Conversion:
console.log(String(undefined)); // "undefined"
console.log(String(null)); // "null"
console.log(String(true)); // "true"
console.log(String(42)); // "42"
console.log(String(NaN)); // "NaN"
console.log(String(Infinity)); // "Infinity"

// Object conversion
let obj = {
    toString() { return "custom string"; }
};
console.log(String(obj)); // "custom string"

// Array conversion
console.log(String([1, 2, 3])); // "1,2,3"
console.log(String([])); // ""


ToBoolean Algorithm:
Converts values to booleans.

Falsy Values: false, 0, -0, 0n, "", null, undefined, NaN
All other values are truthy.

Example - ToBoolean Conversion:
console.log(Boolean(false)); // false
console.log(Boolean(0)); // false
console.log(Boolean(-0)); // false
console.log(Boolean(0n)); // false
console.log(Boolean("")); // false
console.log(Boolean(null)); // false
console.log(Boolean(undefined)); // false
console.log(Boolean(NaN)); // false

// Truthy values
console.log(Boolean(true)); // true
console.log(Boolean(1)); // true
console.log(Boolean(-1)); // true
console.log(Boolean("0")); // true (non-empty string)
console.log(Boolean("false")); // true (non-empty string)
console.log(Boolean({})); // true (object)
console.log(Boolean([])); // true (array)
console.log(Boolean(function() {})); // true (function)


7.2 ABSTRACT EQUALITY (==) ALGORITHM
--------------------------------------------------------------------------------

The loose equality operator performs type coercion according to specific rules.

Equality Algorithm Steps:
1. If types are same, use strict equality (===)
2. null == undefined (and vice versa) → true
3. Number compared with String → convert String to Number
4. Boolean compared with anything → convert Boolean to Number
5. Object compared with primitive → convert Object to primitive
6. Otherwise → false

Example - Abstract Equality Cases:
// Same type: no coercion
console.log(5 == 5); // true
console.log("hello" == "hello"); // true

// null and undefined
console.log(null == undefined); // true
console.log(null == null); // true
console.log(undefined == undefined); // true

// Number and String
console.log(5 == "5"); // true (string "5" → number 5)
console.log(0 == ""); // true (empty string → 0)
console.log(0 == "0"); // true

// Boolean coercion
console.log(true == 1); // true (true → 1)
console.log(false == 0); // true (false → 0)
console.log(true == "1"); // true (true → 1, "1" → 1)
console.log(false == ""); // true (false → 0, "" → 0)

// Object to primitive
let obj = {
    valueOf() { return 42; }
};
console.log(obj == 42); // true (obj → 42)

// Unexpected results
console.log("" == 0); // true
console.log("0" == 0); // true
console.log("" == "0"); // false (both strings, no coercion)
console.log(false == "false"); // false (false → 0, "false" → NaN)
console.log(false == "0"); // true (false → 0, "0" → 0)


7.3 WRAPPER OBJECTS
--------------------------------------------------------------------------------

Primitives are temporarily wrapped in objects when methods are accessed.

Wrapper Constructors:
- Boolean()
- Number()
- String()
- Symbol() (not a constructor)
- BigInt() (not a constructor)

Example - Automatic Wrapping:
// Primitives don't have methods, but JavaScript wraps them temporarily
let str = "hello";
console.log(str.toUpperCase()); // "HELLO"

// Equivalent to:
// let temp = new String("hello");
// let result = temp.toUpperCase();
// temp is discarded

// Wrapper object vs primitive
let primitive = "hello";
let wrapper = new String("hello");

console.log(typeof primitive); // "string"
console.log(typeof wrapper); // "object"

console.log(primitive === "hello"); // true
console.log(wrapper === "hello"); // false (object vs primitive)
console.log(wrapper.valueOf() === "hello"); // true

// Wrapper objects are truthy (even for false)
if (new Boolean(false)) {
    console.log("Wrapper object is truthy!"); // This executes
}

if (false) {
    console.log("Primitive false is falsy"); // This does not execute
}


Example - Explicit Wrapper Creation (Avoid):
// Creating wrapper objects explicitly (not recommended)
let numWrapper = new Number(42);
let strWrapper = new String("hello");
let boolWrapper = new Boolean(true);

console.log(numWrapper + 10); // 52 (auto-unboxed)
console.log(strWrapper.length); // 5
console.log(boolWrapper.valueOf()); // true

// Problem: wrapper objects are always truthy
if (new Boolean(false)) {
    console.log("This executes!"); // Unexpected behavior
}

// Best practice: use constructors without 'new' for type conversion
let num = Number("42"); // 42 (primitive)
let str = String(42); // "42" (primitive)
let bool = Boolean(1); // true (primitive)

8.2 TYPE CHECKING PATTERNS
--------------------------------------------------------------------------------

Example - Robust Type Checking:
// Checking for undefined/null
function processValue(value) {
    if (value === undefined || value === null) {
        return "No value";
    }
    return value;
}

// Nullish coalescing for default values
function greet(name) {
    name = name ?? "Guest";
    return `Hello, ${name}`;
}

// Checking for arrays
function processArray(arr) {
    if (!Array.isArray(arr)) {
        throw new TypeError("Expected array");
    }
    return arr.map(x => x * 2);
}

// Checking for objects (excluding null and arrays)
function isPlainObject(value) {
    return typeof value === "object" 
        && value !== null 
        && !Array.isArray(value);
}

console.log(isPlainObject({})); // true
console.log(isPlainObject([])); // false
console.log(isPlainObject(null)); // false


8.3 AVOIDING COMMON PITFALLS
--------------------------------------------------------------------------------

Example - Floating Point Precision:
// Problem: floating point arithmetic
console.log(0.1 + 0.2); // 0.30000000000000004

// Solution: use epsilon comparison
function areFloatsEqual(a, b, epsilon = Number.EPSILON) {
    return Math.abs(a - b) < epsilon;
}

console.log(areFloatsEqual(0.1 + 0.2, 0.3)); // true

// Or: work with integers
function addCurrency(a, b) {
    // Store cents as integers
    return (a * 100 + b * 100) / 100;
}

console.log(addCurrency(0.1, 0.2)); // 0.3


Example - NaN Checking:
// Problem: NaN is not equal to itself
console.log(NaN === NaN); // false

// Solution: use Number.isNaN()
console.log(Number.isNaN(NaN)); // true
console.log(Number.isNaN("hello")); // false (does not coerce)

// Avoid global isNaN (coerces to number first)
console.log(isNaN("hello")); // true (coerces "hello" to NaN)
console.log(Number.isNaN("hello")); // false (correct)


Example - Avoiding Implicit Coercion:
// Problem: unexpected coercion
console.log("5" + 3); // "53" (string concatenation)
console.log("5" - 3); // 2 (numeric subtraction)

// Solution: explicit conversion
let str = "5";
let num = 3;
console.log(Number(str) + num); // 8
console.log(str + String(num)); // "53"


Example - Safe Property Access:
// Problem: accessing nested properties
let user = {
    profile: {
        address: {
            city: "New York"
        }
    }
};

// Unsafe: throws if intermediate property is undefined
try {
    console.log(user.profile.contact.phone); // TypeError
} catch (error) {
    console.log("Cannot access nested property");
}

// Solution 1: manual checking
let phone = user.profile && user.profile.contact && user.profile.contact.phone;
console.log(phone); // undefined (safe)

// Solution 2: optional chaining (modern JavaScript)
console.log(user.profile?.contact?.phone); // undefined (safe)
console.log(user.profile?.address?.city); // "New York"


8.4 PERFORMANCE CONSIDERATIONS
--------------------------------------------------------------------------------

Example - Avoiding Repeated Type Conversion:
// Inefficient: repeated conversion
function processItems(items) {
    for (let i = 0; i < items.length; i++) {
        console.log(String(items[i])); // Converts each iteration
    }
}

// Efficient: convert once if possible
function processItemsEfficient(items) {
    let stringItems = items.map(String);
    for (let i = 0; i < stringItems.length; i++) {
        console.log(stringItems[i]);
    }
}


Example - Caching Length in Loops:
// Less efficient: accesses length property each iteration
function sumArray(arr) {
    let sum = 0;
    for (let i = 0; i < arr.length; i++) {
        sum += arr[i];
    }
    return sum;
}

// More efficient: cache length
function sumArrayCached(arr) {
    let sum = 0;
    let len = arr.length;
    for (let i = 0; i < len; i++) {
        sum += arr[i];
    }
    return sum;
}
